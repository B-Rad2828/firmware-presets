#$ DESCRIPTION: Basic F411 Setup
#$ DESCRIPTION: - Smart Audio UART 2
#$ DESCRIPTION: - Disable Accelerometer, Barometer & Magnometer
#$ DESCRIPTION: - Vbat & Amp scaling
#$ DESCRIPTION: - Disable unwanted features & beepers
#$ DESCRIPTION: - Set channel mapping
#$ DESCRIPTION: - My AUX switches
#$ DESCRIPTION: - Wider throttle response range
#$ DESCRIPTION: - Motor direction reversed

# ------ F411 SETUP PRESET BEGIN ------

# reset configuration to default settings
defaults nosave

# serial (Smart Audio UART 2)
serial 1 2048 115200 57600 0 115200

# master
set acc_hardware = NONE
set mag_hardware = NONE
set baro_hardware = NONE

# feature
feature -SOFTSERIAL
feature -TELEMETRY

set vbat_scale = 112
set ibata_scale = 1175

# beeper
beeper -GYRO_CALIBRATED
beeper -RX_LOST_LANDING
beeper -DISARMING
beeper -ARMING
beeper -ARMING_GPS_FIX
beeper -BAT_LOW
beeper -GPS_STATUS
beeper -ACC_CALIBRATION
beeper -ACC_CALIBRATION_FAIL
beeper -READY_BEEP
beeper -DISARM_REPEAT
beeper -ARMED
beeper -SYSTEM_INIT
beeper -ON_USB
beeper -BLACKBOX_ERASE
beeper -CRASH_FLIP
beeper -CAM_CONNECTION_OPEN
beeper -CAM_CONNECTION_CLOSE
beeper -RC_SMOOTHING_INIT_FAIL

# beacon
beacon RX_LOST
beacon RX_SET

# map
map TAER1234

# aux (Switches)
aux 0 0 0 1875 2100 0 0
aux 1 13 2 1375 1625 0 0
aux 2 35 2 1900 2100 0 0

# -- Wide min/max throttle
set min_check = 1005
set max_check = 2000

set yaw_motors_reversed = ON

# ------ F411 SETUP PRESET END ------
